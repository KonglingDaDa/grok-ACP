### Mock 工具函数（放在测试文件顶部）

```js
import { describe, it, mock } from "node:test";
import assert from "node:assert";
import { Readable, Writable } from "node:stream";
import { EventEmitter } from "node:events";

/**
 * 创建一个 fake ChildProcess，用于 mock child_process.spawn 返回值。
 * 
 * @returns {Object} fake ChildProcess 实例，包含 stdin/stdout/stderr 流和事件
 */
function createFakeProcess() {
  const proc = new EventEmitter();
  
  // stdin：写入流（测试代码 → Grok）
  proc.stdin = new Writable({
    write(chunk, encoding, callback) {
      // 记录写入的数据，供断言检查
      if (!proc.stdin._written) proc.stdin._written = [];
      proc.stdin._written.push(chunk.toString());
      callback();
    }
  });
  
  // stdout：读取流（Grok → 测试代码）
  proc.stdout = new Readable({
    read() {}  // no-op，由 push() 控制
  });
  
  // stderr：读取流（Grok 错误日志 → 测试代码）
  proc.stderr = new Readable({
    read() {}
  });
  
  // 辅助方法：模拟 Grok 发送 JSON-RPC 响应
  proc._sendLine = (line) => {
    proc.stdout.push(line + '\n');
  };
  
  // 辅助方法：模拟进程退出
  proc._exit = (code, signal = null) => {
    proc.emit('exit', code, signal);
    proc.stdout.push(null);  // EOF
    proc.stderr.push(null);
  };
  
  return proc;
}

/**
 * Mock child_process.spawn，返回 fake process。
 * 
 * @param {Function} setupFn - 设置函数，接收 fake process，可以在里面调用 _sendLine/_exit
 * @returns {Function} mock spawn 函数
 */
function mockSpawn(setupFn) {
  return (cmd, args, options) => {
    const proc = createFakeProcess();
    
    // 异步执行 setup，模拟 Grok 的响应时序
    setImmediate(() => {
      try {
        setupFn(proc, cmd, args, options);
      } catch (err) {
        proc._exit(1);
      }
    });
    
    return proc;
  };
}
```
